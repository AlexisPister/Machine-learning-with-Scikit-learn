Dans notre cas d'tude o on recherche des profils qui peuvent rembourser des crdit, il parait plus judicieux de comparer la prcision que le rappel. En effet, si la banque donne un crdit non rembous, elle est beaucoup plus perdante que si elle gagne un crdit en plus. On cherche donc plus la scurit et donc la prcision que la quantit associe au rappel.

L'arbre CART et les k-plus-proche-voisins donnent respectivement un score-F1 (correspondant ici l'accuracy) pondr par les proportions des 2 classes de 0.71 et 0.70, et une prcision pondre de 0.71 et 0.69.

Les 2 classifieurs donnent donc des rsultats trs proche et il est dur de les diffrencier. A choisir, on favoriserai tout de mme l'arbre CART donnant un score et une prcision un peu plus important.

from sklearn.model_selection import KFold
from sklearn.model_selection import cross_val_score
from sklearn.naive_bayes import GaussianNB
from sklearn.neural_network import MLPClassifier    
from sklearn.ensemble import BaggingClassifier
from sklearn.ensemble import AdaBoostClassifier
import time

# Differents classifieurs
clfs = {
'NB': GaussianNB(),
'CART': DecisionTreeClassifier(random_state=1),
'RF': RandomForestClassifier(n_estimators=50, random_state=1),
'ID3': DecisionTreeClassifier(criterion = 'entropy', random_state=1),
'MLP': MLPClassifier(solver='lbfgs',alpha=1e-5,hidden_layer_sizes=(20,10),random_state=1),
'KPPV': KNeighborsClassifier(n_neighbors=7),
'BAGGING': BaggingClassifier(n_estimators=50,random_state=1),
'ADABOOST': AdaBoostClassifier(n_estimators=50, random_state=1)
}

# Fonction de test des diffrents classifieurs
def run_classifiers(clfs,X,Y):
    Xnorm = MinMaxScaler().fit(X).transform(X)
    acp = PCA(n_components=0.90, svd_solver='full')
    Xacp = acp.fit_transform(Xnorm)
    Xproc = np.hstack((Xnorm, Xacp[:,:]))
    
    kf = KFold(n_splits=10, shuffle=True, random_state=1)
    for i in clfs:
        clf = clfs[i]
        print("\n\n======= {0} =======".format(i))
        start_time = time.time()
        cv_acc = cross_val_score(clf, Xproc, Y, cv=kf)
        print('\nTemps d\'excution d\'un cross val : %s secondes'%(time.time() - start_time))
        cv_auc = cross_val_score(clf, Xproc, Y, cv=kf, scoring='roc_auc')
        cv_prec = cross_val_score(clf, Xproc, Y, cv=kf, scoring='precision')
        print("Accuracy for {0} is: {1:.3f} +/- {2:.3f}".format(i, np.mean(cv_acc), np.std(cv_acc)))
        print("AUC for {0} is: {1:.3f} +/- {2:.3f}".format(i, cv_auc.mean(), cv_auc.std()))
        print("precision for {0} is: {1:.3f} +/- {2:.3f}".format(i, cv_prec.mean(), cv_prec.std()))

run_classifiers(clfs, X, Y)

def run_classifiers_acc(clfs,X,Y):
    X = StandardScaler().fit(X).transform(X)
    kf = KFold(n_splits=3, shuffle=True, random_state=1)
    for i in clfs:
        clf = clfs[i]
        print("\n\n======= {0} =======".format(i))
        start_time = time.time()
        cv_acc = cross_val_score(clf, X, Y, cv=kf)
        print('\nTemps d\'excution d\'un cross val : %s secondes'%(time.time() - start_time))
        print("Accuracy for {0} is: {1:.3f} +/- {2:.3f}".format(i, np.mean(cv_acc), np.std(cv_acc)))
